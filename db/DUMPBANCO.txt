CREATE TABLE USER (ID INTEGER NOT NULL, NAME VARCHAR(30), LOGIN VARCHAR(15), SENHA VARCHAR(15), PRIMARY KEY (ID));
CREATE TABLE CATEGORY (ID INTEGER NOT NULL, DESCRIPTION VARCHAR(50), TYPE VARCHAR(30), PRIMARY KEY (ID));
